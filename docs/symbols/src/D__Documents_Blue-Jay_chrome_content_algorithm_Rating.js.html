 20 /* Name: Rating
 21  * Description: the Rating class is a score that is assigned to a certain 
 22  * Candidate at a certain time
 23  */
 24 
 25 function Rating() {
 26     //public function prototypes
 27     this.setActivityName = setActivityName;
 28     this.getActivityName = getActivityName;
 29     this.setDate = setDate;
 30     this.getDate = getDate;
 31     this.setScore = setScore;
 32     this.getScore = getScore;
 33     this.setWeight = setWeight;
 34     this.getWeight = getWeight;
 35     this.stringVersion = stringVersion;
 36     
 37     //private variables 
 38     var activityName = new Name();
 39     var creationDate = new Date();
 40     var score = 0;
 41     var weight = 1;
 42 
 43     //public functions
 44 	
 45 	// set activity name
 46     function setActivityName(name) {
 47         activityName = name.makeCopy();
 48     }
 49 	// return activity name
 50     function getActivityName() {
 51         return activityName;
 52     }
 53 
 54 	// set date
 55     function setDate(date) {
 56         creationDate = date;
 57     }
 58 	// retun the date when it was created
 59     function getDate() {
 60         return creationDate;
 61     }
 62 
 63 	// set score
 64     function setScore(newValue) {
 65         score = newValue;
 66     }
 67 	// return score
 68     function getScore() {
 69         return score;
 70     }
 71 
 72 	// set weight
 73     function setWeight(newWeight) {
 74         weight = newWeight;
 75     }
 76 	// return weight
 77     function getWeight() {
 78         return weight;
 79     }
 80 
 81     // returns a string representing this Rating
 82     function stringVersion() {
 83         var result = "<Rating>";
 84         result += ("<Activity>" + activityName.getName() + "</Activity>");
 85         result += ("<Score>" + score + "</Score>");
 86         if (creationDate) {
 87             result += ("<Date>" + creationDate.stringVersion() + "</Date>");
 88         }
 89         result += "</Rating>";
 90         return result;
 91     }
 92 };
 93 
 94 // Tells which rating comes first chronologically for the purposes of sorting
 95 function RatingPrecedes() {
 96     //public function prototypes
 97     this.operator = operator;
 98 
 99     //public functions
100     //create a new date equal to this date plus numSeconds
101     function operator(r1, r2) {
102         var t1 = r1;
103         var t2 = r2;
104         // first sort by date
105         if (strictlyChronologicallyOrdered(t1, t2)) {
106             return true;
107         }
108         if (strictlyChronologicallyOrdered(t2, t1)) {
109             return false;
110         }
111         // break ties by name
112         return r1.getActivityName() < r2.getActivityName();
113     }
114 };
