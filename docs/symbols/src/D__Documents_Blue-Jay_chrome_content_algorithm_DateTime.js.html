 20 /** @class simple class that holds date and time */
 21 function DateTime(yyyymmddhhmmss) {
 22     /**
 23      * alert("constructing date");
 24      * public function prototypes
 25      */
 26     this.datePlusDuration= datePlusDuration;
 27     this.stringVersion= stringVersion;
 28     this.timeUntil = timeUntil;
 29     this.setComponents = setComponents;
 30     this.setDurationSinceReference = setDurationSinceReference;
 31     this.getDurationSinceReference = getDurationSinceReference;
 32     this.setNow = setNow;   /** makes the date equal to the current date
 33     this.setInternalValue = setInternalValue;
 34 
 35     /**private variables */
 36     var value = new Date();
 37     /**private functions */
 38     setComponents(yyyymmddhhmmss);
 39 
 40     /** public functions */
 41     /** create a new date equal to this date plus numSeconds */
 42     function datePlusDuration(numSeconds) {
 43         /**
 44          * var duration = new Date(numSeconds*1000);
 45          * alert("new Date()");
 46          */
 47         var newValue = new Date(value.getTime() + numSeconds*1000);
 48         /** alert("new DateTime()"); */
 49         var result = new DateTime();
 50         /** alert("setInternalValue"); */
 51         result.setInternalValue(newValue); /** this should be a private function */
 52         /**
 53          * alert("my duration " + this.getDurationSinceReference());
 54          * alert("other duration " + result.getDurationSinceReference());
 55          * alert("timeUntil");
 56          */
 57         var duration = this.timeUntil(result);
 58         /**
 59          * alert("checking equality");        
 60          * if (duration != numSeconds) {
 61          *     alert(duration + " not equal to " + numSeconds);
 62          * }
 63          * alert("returning result");
 64          */
 65         return result;
 66     }
 67 
 68     /** returns a printable string that represents this date */
 69     function stringVersion() {
 70         var result = "";
 71         result += value.getFullYear();
 72         result += "-";
 73         result += value.getMonth();
 74         result += "-";
 75         result += value.getDate();
 76         result += "T";
 77         result += value.getHours();
 78         result += ":";
 79         result += value.getMinutes();
 80         result += ":";
 81         result += value.getSeconds();
 82         return result;
 83         /** return value.toString(); */
 84     }
 85 
 86     /** duration is in seconds */
 87     function getDurationSinceReference() {
 88         return value.valueOf() / 1000.0;
 89     }
 90     
 91     /** duration is in seconds */
 92     function setDurationSinceReference(numSeconds) {
 93         value = new Date(numSeconds * 1000);
 94     }
 95     /** the number of seconds from this to other */
 96     function timeUntil(other) {
 97         /** alert("::timeUntil"); */
 98         var myDuration = this.getDurationSinceReference();
 99         /** alert("myDuration = " + myDuration); */
100         var otherDuration = other.getDurationSinceReference();
101         /** alert("otherDuration = " + otherDuration); */
102         var result = otherDuration - myDuration;
103         /** alert("result = " + result); */
104         return result;
105     }
106     
107     /** make this DateTime object equal to the current date */
108     function setNow() {
109         value = new Date();
110     }
111     /** private functions */
112 
113     /**
114      * assign the given datestring to this date. An example
115      * datestring would be "1970-00-01T01:01:01"
116      */
117     function setComponents(yyyymmddhhmmss) {
118         if(!yyyymmddhhmmss) {
119             yyyymmddhhmmss = "1970 00 01 00 00 00";
120         }
121         var i;
122         /** replace any kind of separator with spaces */
123         yyyymmddhhmmss = yyyymmddhhmmss.replace("-", " ");
124         yyyymmddhhmmss = yyyymmddhhmmss.replace("-", " ");
125         yyyymmddhhmmss = yyyymmddhhmmss.replace("T", " ");
126         yyyymmddhhmmss = yyyymmddhhmmss.replace(":", " ");
127         yyyymmddhhmmss = yyyymmddhhmmss.replace(":", " ");
128         /**
129          * split it into pieces and assign
130          * the date components accordingly
131          */
132         var newTime = yyyymmddhhmmss.split(" ");
133         value.setYear(newTime[0]);
134         value.setMonth(newTime[1]);
135         value.setDate(newTime[2]);
136         value.setHours(newTime[3]);
137         value.setMinutes(newTime[4]);
138         value.setSeconds(newTime[5]);
139     }
140     function setInternalValue(newValue) {
141         value = newValue;
142     }
143     /** alert("done constructing date"); */
144 };
145 
146 /**
147  * returns true if t1 comes before t2, and
148  * false if t2 is equal to or after t1
149  */
150 function strictlyChronologicallyOrdered(t1, t2) {
151     if (t1.timeUntil(t2)>0) {
152         return true;
153     }
154     return false;
155 }
