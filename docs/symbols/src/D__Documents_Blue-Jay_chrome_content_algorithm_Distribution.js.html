 20 /** @class the Distribution class represents a summary of a collection
 21  * of floating-point numbers it stores average, spread, and count
 22  * (although even count may be fractional)
 23  */
 24 function Distribution(average, stdDev, relativeWidth) {
 25 	/** public function prototypes */
 26 	this.getMean = getMean;
 27 	this.getStdDev = getStdDev;
 28 	this.getWeight = getWeight;
 29 	
 30 	/** private variables */
 31 	var mean = average;
 32 	var standardDeviation = stdDev;
 33 	var weight = relativeWidth;
 34 	
 35 	/** public functions */
 36 	
 37 	/** returns mean value */
 38 	function getMean() {
 39 		return mean;
 40 	}
 41 	
 42 	/** returns standard deviation */
 43 	function getStdDev() {
 44 		return standardDeviation;
 45 	}
 46 
 47 	/** returns weight */
 48 	function getWeight() {
 49 		return weight;
 50 	}
 51 };
