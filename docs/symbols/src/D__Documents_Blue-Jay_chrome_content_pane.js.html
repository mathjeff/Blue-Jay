149   // this function scans the user's library and send that data to the engine
150   scanLibrary : function() {
151     var list = LibraryUtils.mainLibrary;
152     var mystring = "";
153     // iterate over each thing in the library
154     alert("push [ok] to start scanning library");
155     var length = list.length;
156     // limit the number of songs (for testing)
157     // if (length > 65) {
158     //    length = 65;
159     // }
160     var i;
161     var music = new Name("Song");
162     for (i = 0; i < length; i++) {
163         var item = list.getItemByIndex(i);
164         // get the song name
165         var songName = new Name(item.getProperty(SBProperties.trackName));
166         // make sure the song name is something sensible
167         if (!songName.isNull()) {
168             // make a new candidate to represent the song
169             var songCandidate = new Candidate();
170             songCandidate.setName(songName);
171             songCandidate.setPlayable(true);
172             // get the date it was added to the library
173             var timeString = item.getProperty(SBProperties.created);
174             var deltaTime = parseInt(item.getProperty(SBProperties.created));
175             var discoveryDate = new DateTime();
176             discoveryDate.setDurationSinceReference(deltaTime / 1000);
177             songCandidate.setDiscoveryDate(discoveryDate);
178             message("getting rating");
179             // get the song's rating if it has been rated yet
180             var starCount = item.getProperty(SBProperties.rating);
181             message(starCount + "\r\n");
182             if (starCount > 0) {
183                 // create the rating and add it
184                 var songRating = new Rating();
185                 songRating.setActivityName(songName);
186                 songRating.setScore((starCount - 1) / 4);
187                 // setting an invalid date means we want it to be the earliest possible date
188                 songRating.setDate(null);
189                 this.engine.putRatingInMemory(songRating);
190             }
191 
192 
193             // get the artist name
194             var artistName = new Name(item.getProperty(SBProperties.artistName));
195             // if the artist name is unknown, group it with other songs with unknown genre
196             if (artistName.isNull()) {
197                 artistName = new Name("Unspecified Artist");
198             }
199             // create a Candidate representing the artist
200             var artistCandidate = new Candidate();
201             artistCandidate.setName(artistName);
202             artistCandidate.addParentName(music);
203             artistCandidate.setPlayable(false);
204             // give the artist candidate to the engine
205             this.engine.addCandidate(artistCandidate);
206             // add the artist as a parent of the song
207             songCandidate.addParentName(artistName);
208 
209 
210             // get the genre
211             var genre = new Name(item.getProperty(SBProperties.genre));
212             // if the genre is unknown, group it with other songs with unknown genre
213             if (genre.isNull()) {
214                 genre = new Name("Unspecified Genre");
215             }
216             // create a Candidate representing the genre
217             var genreCandidate = new Candidate();
218             genreCandidate.setName(genre);
219             genreCandidate.addParentName(music);
220             genreCandidate.setPlayable(false);
221             // give the genre candidate to the engine
222             this.engine.addCandidate(genreCandidate);
223             // add the artist as a parent of the song
224             songCandidate.addParentName(genre);
225             
226             /* // if the artist name and genre name are unknown, then the song is a direct child of the category "Song"
227             if (artistName.isNull() && genre.isNull()) {
228                 songCandidate.addParentName(music);
229             }*/ 
230             
231             // give the song to the engine
232             this.engine.addCandidate(songCandidate);
233         }
234     }
235     var musicCategory = new Candidate();
236     musicCategory.setName(music);
237     musicCategory.setPlayable(false);
238     this.engine.addCandidate(musicCategory);    
239     this.isLibraryScanned = true;
240     //alert("done scanning library");
241     // read any data from the text files that gives information about the library
242     this.engine.readFiles();
243     flushMessage();
244   },

246   // this function gets called whenever the song changes
247   songChanged: function(ev) {
248         this.clearRatingMenu();
249 		// get the data for the new track
250 		var mediaItem = ev.data;
251 		var songName = mediaItem.getProperty(SBProperties.trackName)
252 		//alert("Selected track: \"" + songName + "\" by " + mediaItem.getProperty(SBProperties.artistName));
253 		var songLength=mediaItem.getProperty(SBProperties.duration)/1000000;
254 		// get the current date
255 		this.songEndDate = new DateTime();
256 		this.songEndDate.setNow();
257 		// check if we were previously playing a song
258 		if (this.currentSongName && (this.currentSongName != this.ignoredSongname)) {
259 			// if we get here then we were previously playing a song
260 			// compute the duration it actually played
261 			var playedDuration = this.songStartDate.timeUntil(this.songEndDate);
262 			//alert("played duration = " + playedDuration + " song length = " + this.currentSongDuration);
263 			// decide whether it was skipped based on the duration
264 			if (playedDuration >= this.currentSongDuration * 0.75) {
265 				// if we get here then it was not skipped
266 				var newParticipation = new Participation();
267 				newParticipation.setStartTime(this.songStartDate);
268 				newParticipation.setEndTime(this.songEndDate);
269 				newParticipation.setIntensity(1);
270 				newParticipation.setActivityName(new Name(this.currentSongName));
271 				this.engine.addParticipation(newParticipation);
272 			} else {
273 				// if we get here then the song was skipped
274 				var newRating = new Rating();
275 				newRating.setActivityName(new Name(this.currentSongName));
276 				newRating.setDate(this.songEndDate);
277 				newRating.setScore(0);
278 				this.engine.addRating(newRating);
279 			}
280 		}
281 		// update the current song
282 		this.currentSongName = songName;
283 		this.songStartDate = this.songEndDate;
284 		this.currentSongDuration = songLength;
285 		//alert("You have skipped this item " + mediaItem.getProperty(SBProperties.skipCount) + " times and its duration is " + songLength + " seconds");
286 		if (songName != this.desiredTrackName) {
287 			// if a song was chosen randomly, and we then skip it automatically, that's not a downvote
288 			this.ignoredSongname = songName;
289 			// only change the song if the user wants us to
290 			if (this.state == "on") {
291     			this.makePlaylist();
292             }
293 		}
294   },

332         //var tracks = LibraryUtils.mainLibrary.getItemsByProperty(SBProperties.artistName, "Dexys Midnight Runners");
