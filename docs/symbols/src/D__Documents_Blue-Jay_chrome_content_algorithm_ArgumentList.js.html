 20 /** @class took argument list from the command line (depreciated) */
 21 
 22 function ArgumentList(argc, argv) {
 23 	/** public function prototypes */
 24 	this.getNumArguments = getNumArguments;
 25 	this.getArgument = getArgument;
 26 	
 27 	/** private variables */
 28 	var numArguments = argc | 0;
 29 	var argumentValues = argv | {};
 30 	
 31 	/** public functions */
 32 	function getNumArguments() {
 33 		return numArguments;
 34 	}
 35 	
 36 	function getArgument(index) {
 37 		return argumentValues[index];
 38 	}
 39 	
 40 	
 41 };
