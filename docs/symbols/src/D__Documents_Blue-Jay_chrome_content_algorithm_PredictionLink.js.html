 20 /* Name: PredictionLink
 21  * Description: The PredictionLink class is used to estimate the rating 
 22  * that will be given to another song (or category) based on an attribute 
 23  * of another song (or category)
 24  */
 25 
 26  function PredictionLink(passedVal1, passedVal2) {
 27     //alert("constructing predictionLink\r\n");
 28 	
 29 	/* public function */
 30 	this.initializeDecreasing = initializeDecreasing;
 31 	this.initializeIncreasing = initializeIncreasing;
 32 	this.update = update;
 33 	this.guess = guess;
 34 	
 35 	
 36 	//private variables
 37 	var inputData = passedVal1;
 38 	var outputData = passedVal2;
 39     //alert("constructing Scatterplot\r\n");
 40 	var plot = new ScatterPlot();
 41     //alert("constructing DateTime\r\n");
 42 	var previousNumRatings = 0;
 43 	var numChanges = 0.0;
 44 	
 45 	//message("making prediction link part 1\r\n");
 46 	// check whether this link is predicting something from its own past history
 47 	if (inputData.getOwnerName().equalTo(outputData.getOwnerName())) {
 48 	
 49 	    //message("making prediction link part 2\r\n");
 50 		// check whether this link is using the participation history
 51 		if (inputData.isAParticipationMovingAverage()) {
 52     	    //message("making prediction link part 3\r\n");
 53 			
 54 			// If we get here then we're predicting the score of a Candidate based
 55 			// on its past frequency. Usually, if something has happened a lot recently 
 56 			// then it will be boring in the future
 57 			//this.initializeDecreasing();
 58 		} else {
 59 			// If we get here then we're predicting the score of a Candidate based
 60 			// on its current score. If we start with a small amount of suspicion that it will
 61 			// be positively correlated then we may save the user lots of time
 62 			// We use a small weight, though, so it's easy to overpower
 63 		    this.initializeIncreasing();
 64 		}
 65 	}
 66 	
 67 
 68 
 69 	//public functions
 70 	// initialize this PredictionLink with the assumption that the data is negatively correlated
 71 	function initializeDecreasing() {
 72 	    //message("making prediction link part 4\r\n");
 73 	
 74 		var intensity = 1;
 75 		var numPoints = 40;
 76 		var i = 0;
 77 		var score = 0.0;
 78 		var duration = 0.0;
 79 		
 80 		for (i = 0; i < numPoints; i++) {
 81     		
 82 			/*duration = i*1500.0;
 83 			intensity = 1.0/duration;
 84 			score = Math.sqrt(duration) / 250.0;
 85 			*/
 86 			duration = i * 1500;
 87 			intensity = 1 / duration;
 88 			score = i / numPoints;
 89 			
 90 			// add some extra variation to show that we aren't sure about this
 91 			if (i % 2 == 0)
 92 			    score = (1 + score) / 2;
 93 			else
 94 			    score = score / 2;
 95 			plot.addDataPoint(new Datapoint(intensity, score, 1));
 96 		}
 97 		numChanges = numChanges + numPoints;
 98 	}
 99 	
100 	// initialize this PredictionLink with the assumption that the data is positively correlated
101 	function initializeIncreasing() {
102 	    plot.addDataPoint(new Datapoint(0, 0, 1));
103 	    plot.addDataPoint(new Datapoint(0, 0, 1));
104 	    plot.addDataPoint(new Datapoint(0, 0, 1));
105 	    plot.addDataPoint(new Datapoint(.25, .25, 1));
106 	    plot.addDataPoint(new Datapoint(.5, .5, 1));
107 	    plot.addDataPoint(new Datapoint(.75, .75, 1));
108 	    // We want at least sqrt(n) points with x=y=1 so that when input=1 we predict output=1
109 	    plot.addDataPoint(new Datapoint(1, 1, 1));
110 	    plot.addDataPoint(new Datapoint(1, 1, 1));
111 	    plot.addDataPoint(new Datapoint(1, 1, 1));
112 	    // numChanges is a measure of how certain we are. It doesn't need to equal the number of data points
113 	    numChanges += 4;
114 	}
115 	
116 	// updates the scatterplot with any new data that it hadn't yet 
117 	// requested from the MovingAverage that it tries to estimate
118 	function update() {
119         //alert("PredictionLink::update\r\n");
120 		var newPoints = inputData.getCorrelationsFor(outputData, previousNumRatings);
121         //alert("back in PredictionLink::update\r\n");
122 		
123         //message("plot adding " + newPoints[0].length + " datapoints\r\n");
124 		if (newPoints[0].length > 0) {
125 		
126 			var i=0;
127 			for (i = 0; i < newPoints[0].length; i++) {
128 				plot.addDataPoint(newPoints[0][i]);
129 			}
130 		
131 			latestUpdateTime = outputData.getLatestDate();
132 			numChanges = numChanges + newPoints[1];
133             //alert("plot done adding datapoints\r\n");
134 		}
135 		previousNumRatings = outputData.getNumRatings();
136 	}
137 	
138 	// compute a distribution that represents the expected deviation from the overall mean
139 	function guess(when) {
140 	    // if we don't have any data...
141 	    if (inputData.getIndexForDate(when, false) < 0) {
142 	        // then we should return a guess with no weight
143 	        return new Distribution(0, 0, 0);
144 	    }
145 		//alert("PredictionLink::guess");
146 
147         // get the current value of the input. This value is a distribution, not a float		
148 		var input = inputData.getCurrentValue(when, false);
149 		// make a guess for this input
150 		var middle = plot.predict(input.getMean());
151 		
152 		message("PredictionLink input = "+input.getMean());
153 		message("middle = " + middle.getMean());
154 		
155 	    var stdDevA = middle.getStdDev();
156 	    var stdDevB = 0;
157 	    // If there is uncertainty in the input, then the slope of the line affects the output uncertainty
158 		if (input.getStdDev() > 0) {
159 		    var leftOneStdDev = plot.predict(input.getMean() - input.getStdDev());
160 		    message(" left output = " + leftOneStdDev.getMean());
161     		
162 		    var rightOneStdDev = plot.predict(input.getMean() + input.getStdDev());
163 		    message(" right output = " + rightOneStdDev.getMean());
164     		
165 		    //alert("PredictionLink::guess pt2");
166     		stdDevB = (rightOneStdDev.getMean() - leftOneStdDev.getMean()) / 2.0;
167         }
168 	    var stdDev = Math.sqrt(stdDevA * stdDevA + stdDevB * stdDevB);
169 
170 		//alert("PredictionLink::guess pt3");
171 		
172 		//var weight = numChanges - 1.0;
173 		var weight = numChanges;
174 		if (weight < 0.0) {
175 			weight = 0.0;
176 		}
177 		var stdDev2;
178 		if (numChanges > 0)
179 		    stdDev2 = .5 / numChanges;
180 		else
181     		stdDev2 = 0.5;
182     				
183 		var result = new Distribution(middle.getMean(), stdDev + stdDev2, weight);
184 		//alert("PredictionLink::guess pt4");
185 		return result;
186 	}
187 }
