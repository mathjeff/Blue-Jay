 20 /**
 21  * @class The Candidate class represent item such as a song, artist,
 22  * genre or other categories. All these items can have ratings.
 23  */
 24 function Candidate(passedVal) {
 25 
 26 	/** public function */
 27 	/** Name */
 28 	this.setName = setName;
 29 	this.getName = getName;
 30 	
 31 	/** unique identifier */
 32 	this.setID = setID;
 33 	
 34 	/**
 35         * whether this represents an actual music track that can
 36 	 * played, or a genre that can't be played directly
 37 	 */
 38     this.isPlayable = isPlayable;
 39     this.setPlayable = setPlayable;
 40     	
 41 	/** adding parent links */
 42 	this.addParentName = addParentName;
 43 	this.addParent = addParent;
 44 	
 45 	/** adding child links */
 46 	this.addChild = addChild;
 47 	this.getParentNames = getParentNames;
 48 	this.getParents = getParents;
 49 	this.getChildren = getChildren;
 50 	this.getNumChildren = getNumChildren;
 51 	
 52 	/** informing the Candidate whenever the user rates it */
 53 	this.giveRating = giveRating;
 54 	
 55 	/** informing the Candidate when it is listened to */
 56 	this.giveParticipation = giveParticipation;
 57 	
 58 	/**
 59         * how many MovingAverages there are to estimate its rating.
 60 	 * Currently it's always 1
 61         */
 62 	this.getNumRatingEstimators = getNumRatingEstimators;
 63 	
 64 	/** get the appropriate MovingAverage */
 65 	this.getRatingEstimatorAtIndex = getRatingEstimatorAtIndex;
 66 	
 67 	/** get the appropriate MovingAverage for different index */
 68 	this.getFrequencyHistory = getFrequencyHistory;
 69 	
 70        /** get the MovingAverage that stores the ratings */
 71 	this.getRatingHistory = getRatingHistory;
 72 	
 73 	/**
 74         * the current expected rating, based
 75         * on data from other Candidates
 76         */
 77 	this.getCurrentRating = getCurrentRating;
 78 	this.setCurrentRating = setCurrentRating;
 79 
 80        /** the latest date for which a rating was assigned	*/
 81 	this.getLatestRatingDate = getLatestRatingDate;
 82 		
 83 	/** whether the parent pointers are up to date */
 84 	this.needToUpdateParentPointers = needToUpdateParentPointers;
 85 	
 86 	/** the date it was added to your library */
 87 	this.setDiscoveryDate = setDiscoveryDate;
 88 	this.getDiscoveryDate = getDiscoveryDate;
 89 	this.suspectDiscoveryDate = suspectDiscoveryDate;
 90 
 91 	/**
 92         * the duration (in seconds) between the 
 93         * latest listening and the time at 'when'
 94         */
 95 	this.getIdleDuration = getIdleDuration;
 96 	this.getDurationSinceLastPlayed = getDurationSinceLastPlayed;
 97 	// the average of all given ratings 
 98 	this.getAverageRating = getAverageRating;
 99 	
100 	
101 	/* private functions */
102 	var initialize;
103 	
104 	/** private variables */
105 	var name = passedVal;
106 	var ID;
107 	var playable = true;
108 	
109 	/**
110         * parentNames is only used at the beginning for storing
111 	 * names before the parents exist
112 	 */
113 	var parentNames = [];
114 	
115 	/**
116         * 'parents' holds pointers to the parents themselves.
117         * It is faster to use than parentNames because removes
118         * the need for another dictionary lookup
119        */
120 	var parents = [];
121 	var children = [];
122 	var frequencyHistory = new ParticipationMovingAverage();
123 	var ratingHistory = new RatingMovingAverage();
124 	/** the latest date at which there was an interaction */
125 	var latestInteractionDate = new DateTime();
126 	var lastPlayDate = new DateTime();
127 	
128 	/**
129         * the rating based on PredictionLinks before incorporating
130         * information about the parents
131         */
132 	var currentRating = new Distribution();	
133 	var latestRatingDate = new DateTime();	
134 	var parentLinksNeedUpdating = false;
135 	var discoveryDate;
136     
137 	/** call the constructor */
138 	if (name) {
139     	initialize();
140     }
141 	
142 	/** public functions */
143 	
144 	/**
145         * Setting the name of the Candidate must be done before
146         * assigning any ratings or participations
147         */
148 	function setName(newName) {
149 		name = newName.makeCopy();
150 		/**
151         * by default, the ID is the same as the name but
152         * may be given a different value
153        */
154 		if (!ID) {
155 		    ID = name;
156 		}
157 		initialize();
158 	}
159 	
160 	/** unique identifier */
161 	function setID(newID) {
162 	    ID = newID;
163 	}
164 	function getID() {
165 	    return ID;
166     }
167 
168     function isPlayable() {
169         return playable;
170     }
171 	/** parent/child connections */
172 	function getName() {
173 		return name;
174 	}
175 	
176 	/** tell the candidate that it has a parent with this name */
177 	function addParentName(newName) {
178 		parentNames.push(newName);
179 		parentLinksNeedUpdating = true;
180 	}
181 	
182 	/** adding a parent */
183 	function addParent(newCandidate) {
184 		parents.push(newCandidate);
185 	}
186 	
187 	/** adding child */
188 	function addChild(newChild) {
189 	    playable = false;
190 		children.push(newChild);
191 	}
192 	function setPlayable(canBePlayed) {
193 	    playable = canBePlayed;
194 	}
195 	
196 	/** get parent names */
197 	function getParentNames() {
198 		return parentNames;
199 	}
200 	
201 	/** get parents */
202 	function getParents() {
203 		return parents;
204 	}
205 	
206 	/** get children */
207 	function getChildren() {
208 		return children;
209 	}
210 	
211 	/** get number of children */
212 	function getNumChildren() {
213 	    return children.length;
214 	}
215 	
216 	/** inform the candidate that the user gave it this rating */
217 	function giveRating(rating)	{
218 	    // record the rating
219 		var i = 0;
220 		/*for (i = 0; i < ratingEstimators.length; i++) {
221 			ratingEstimators[i].addRating(rating);
222 		}*/
223 		ratingHistory.addRating(rating);
224 
225 		/** update the latest interaction date */
226 		latestInteractionDate = rating.getDate();
227 		
228 		/**
229  * If we don't know when we were discovered, then
230  * assume we were discovered along with the first
231  * rating.  If we think we were discovered after
232  * this rating, then it is actually more reasonable
233  * to say we were discovered when this rating was created
234  */
235 		if ((!discoveryDate) || (strictlyChronologicallyOrdered(rating.getDate(), discoveryDate))) {
236 		    discoveryDate = rating.getDate();
237 		}
238 	}
239 	
240 	/** inform the Candidate that it was listened
241         * to during a certain interval
242         */
243 	function giveParticipation(participation) {
244 		var i = 0;
245 		/** update the moving averages of the ratings */
246 		/*for (i = 0; i < frequencyEstimators.length; i++) {
247 			frequencyEstimators[i].addParticipationInterval(participation);
248 		}*/
249 		frequencyHistory.addParticipationInterval(participation);
250 		lastPlayDate = latestInteractionDate = participation.getEndTime();
251 
252 		/** 
253         * If we don't know when we were discovered, then assume
254         * we were discovered along with the first participation.
255         * If we think we were discovered after this rating, then
256         * it is actually more reasonable to say we were discovered
257         * when this rating was created
258         */
259 		if ((!discoveryDate) || (strictlyChronologicallyOrdered(participation.getStartTime(), discoveryDate))) {
260 		    discoveryDate = participation.getStartTime();
261 		}
262 	}
263 	
264 	/**
265         * returns the number of MovingAverages that try
266         * to estimate the current rating
267         */
268 	function getNumRatingEstimators() {
269 		return ratingEstimators.length;
270 	}
271 	
272 	/** returns a particular rating estimator */
273 	function getRatingEstimatorAtIndex(index) {
274 		return ratingEstimators[index];
275 	}
276 		
277 	/**
278         * returns the ParticipationMovingAverage
279         * that keeps track of listening
280         */
281 	function getFrequencyHistory() {
282 		return frequencyHistory;
283 	}
284 	
285 	/** returns the moving average that records the ratings */
286 	function getRatingHistory() {
287 		return ratingHistory;
288 	}
289 	
290 	/** returns current rating */
291 	function getCurrentRating() {
292 		return currentRating;
293 	}
294 	
295 	/** set current rating */
296 	function setCurrentRating(value, when) {
297 		currentRating = value;
298 		latestRatingDate = when;
299 	}
300 
301        /** get the latest date for which a rating was assigned */
302 	function getLatestRatingDate() {
303 	    return latestRatingDate;
304 	}
305 			
306 	function needToUpdateParentPointers() {
307 		if(parentLinksNeedUpdating == true) {
308 			parentLinksNeedUpdating = false;
309 			return true;
310 		}
311 		return false;
312 	}
313 	
314 	/**
315         * set the estimate for the date at which this Candidate was
316         * discovered.  Returns true if the estimate is believable,
317         * false if the estimate is not believable
318         */
319 	function suspectDiscoveryDate(when) {
320 		/**
321         * Internally we estimate the discovery date based on the
322  * first time this song was played or rated.  So, we only
323  * use outside suspicion of the discovery date if it does
324  * not move the discovery date later
325  */
326 	    if ((!discoveryDate) || (strictlyChronologicallyOrdered(when, discoveryDate))) {
327 	        this.setDiscoveryDate(when);
328 	        return true;
329         }
330         return false;
331 	}
332 	/**
333  * declares that this Candidate was discovered that the given date
334  */
335 	function setDiscoveryDate(when) {
336        /**
337         * update the latest date at which there was an interaction
338         */
339        if (strictlyChronologicallyOrdered(latestInteractionDate, when)) {
340        latestInteractionDate = when;
341        }
342        discoveryDate = when;
343 	}
344 	/**
345         * get the estimate for the date at which
346         * this Candidate was discovered
347        */
348        function getDiscoveryDate(when) {       /** if discoveryDate is valid... */
349        if (discoveryDate) {
350        /** ...then return it */
351        return discoveryDate;
352        }
353        else {
354        /**
355         * If we get here, then we don't have any data
356         * suggesting a discovery date.  So, assume it
357         * was discovered an extremely long time agon
358         */
359        return new DateTime();
360 		}
361 	}
362 	
363 	/** Tells how long it has been since the song was heard or rated */
364 	function getIdleDuration(when) {
365 		return latestInteractionDate.timeUntil(when);
366 	}
367 	function getDurationSinceLastPlayed(when) {
368 		return lastPlayDate.timeUntil(when);
369 	}
370 	
371 	function getAverageRating() {
372 		return ratingHistory.getAverageValue();
373 	}
374 	
375 	function initialize() {
376 		frequencyHistory.setName(new Name(name.getName() + " (participations)"));
377 		frequencyHistory.setOwnerName(name);
378 
379 		ratingHistory.setName(new Name(name.getName() + " (ratings)"));
380 		ratingHistory.setOwnerName(name);
381 	}
382 	
383 };
