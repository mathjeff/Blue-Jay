 67 	function addRating(rating) {
 68 		var totalScore = 0.0; 
 69 		var totalWeight = 0.0;
 70 		var totalSquaredScore = 0.0;
 71 		
 72 		if(sumRatings.length > 0) {
 73 			var lastRating = sumRatings[sumRatings.length - 1];
 74 			totalScore = lastRating.getScore();
 75 			totalWeight = lastRating.getWeight();
 76 			var sumSquaredRating = sumSquaredRatings[sumSquaredRatings.length - 1];
 77 			totalSquaredScore = sumSquaredRating.getScore();
 78 		} else {
 79 			totalScore = totalWeigh = totalSquaredScore = 0;
 80 		}
 81 		
 82         //alert("rating moving average adding rating pt2\r\n");	
 83 		// compute new running total
 84 		var newTotalScore = totalScore + rating.getScore() * rating.getWeight();
 85 		var newTotalWeight = totalWeight + rating.getWeight();
 86 		var newRating = new Rating();
 87 		newRating.setScore(newTotalScore);
 88 		newRating.setWeight(newTotalWeight);
 89 		newRating.setDate(rating.getDate());
 90 		
 91 		// save the new running total
 92 		sumRatings.push(newRating);
 93         //alert("rating moving average adding rating pt3\r\n");	
 94 		
 95 		// compute new running squared total
 96 		var newTotalSquaredScore = totalSquaredScore + rating.getScore() * rating.getScore() * rating.getWeight();
 97 		var newSquaredRating = new Rating();
 98 		newSquaredRating.setScore(newTotalSquaredScore);
 99 		newSquaredRating.setWeight(newTotalWeight);
100 		newSquaredRating.setDate(rating.getDate());
101 		sumSquaredRatings.push(newSquaredRating);
102 		
103 		ratings.push(rating);
104         //alert("rating moving average adding rating pt4\r\n");	
105 	}

107     // if strictlyEarlier is true, then it will only use data from strictly before 'when'
108 	function getValueAt(when, strictlyEarlier) {
109 		//alert("RatingMovingAverage::getValueAt pt a\r\n");
110 		message("number of ratings = " + ratings.length + "\r\n");
111 		if (sumRatings.length == 0) {
112 			return [new Distribution(0, 0, 0), -1];
113 		}
114 		
115 		//alert("RatingMovingAverage::getValueAt pt b\r\n");
116 		var firstRating = sumRatings[0];
117 		if(!strictlyChronologicallyOrdered(firstRating.getDate(), when)) {
118     		//alert("RatingMovingAverage::getValueAt pt c\r\n");
119 			return [new Distribution(0, 0, 0), -1];
120 		}
121 		//alert("RatingMovingAverage::getValueAt pt d\r\n");
122 		
123 		var latestRatingIndex = getIndexForDate(when, strictlyEarlier);
124 		//alert("index = " + latestRatingIndex);
125 		var latestRatingDate = sumRatings[latestRatingIndex].getDate();
126 		//alert("calculating duration");
127 		var duration = latestRatingDate.timeUntil(when);
128 		//alert("constructing new date");
129 		var oldestRatingDate = latestRatingDate.datePlusDuration(-duration);
130 		//alert("getting new index");
131 		var earliestRatingIndex = getIndexForDate(oldestRatingDate, true);
132 		//alert("array lookups");		
133 		var latestSumRating = sumRatings[latestRatingIndex];
134 		var latestSumSquaredRating = sumSquaredRatings[latestRatingIndex];
135 		var earliestSumRating = sumRatings[earliestRatingIndex];
136 		var earliestSumSquaredRating = sumSquaredRatings[earliestRatingIndex];
137 		
138 		var sumY = 0.0;
139 		var sumY2 = 0.0;
140 		var sumWeight = 0.0;
141 		//alert("in the middle of RatingMovingAverage::pt e\r\n");
142 		
143 		if(strictlyChronologicallyOrdered(oldestRatingDate, sumRatings[0].getDate())) {
144 			sumY = latestSumRating.getScore();
145 			sumY2 = latestSumSquaredRating.getScore();
146 			sumWeight = latestSumRating.getWeight();
147 		}
148 		else{
149 			sumY = latestSumRating.getScore() - earliestSumRating.getScore();
150 			sumY2 = latestSumSquaredRating.getScore() - earliestSumSquaredRating.getScore();
151 			sumWeight = latestSumSquaredRating.getWeight() - earliestSumSquaredRating.getWeight();
152 		}
153 		
154 		var average = sumY/sumWeight;
155 		var stdDev = Math.sqrt((sumY2 - sumY * sumY/sumWeight)/sumWeight);
156 		var result = new Distribution(average, stdDev, sumWeight);
157 		
158 		//alert("done with RatingMovingAverage::getValueAt\r\n");
159 		return [result, latestRatingIndex];
160 	}

173     // returns the index of the latest rating before the given date ("when")	
174 	function getIndexForDate(when, strictlyEarlier) {
175 		if(sumRatings.length < 1) {
176 			return -1;
177 		}
178 		
179 		// check whether the target date is before the earliest date for which we have data
180 		var firstDate = sumRatings[0].getDate();
181 		if (strictlyEarlier) {
182 		    if (!strictlyChronologicallyOrdered(firstDate, when)) {
183 		        return -1;
184 		    }
185         } else {
186 		    if (strictlyChronologicallyOrdered(when, firstDate)) {
187 		        return -1;
188 		    }
189         }
190 		
191 
192 		// check whether the target date is after the most recent date
193         //alert("RatingMovingAverage::getIndexForDate nonzero");
194 		var finalDate = sumRatings[sumRatings.length - 1].getDate();
195         //alert("RatingMovingAverage::getIndexForDate comparing last");
196 		if(strictlyChronologicallyOrdered(finalDate, when)) {
197             //alert("RatingMovingAverage::getIndexForDate was last");
198 			return sumRatings.length - 1;
199 		}
200 
201         //alert("RatingMovingAverage::getIndexForDate not last");
202 		
203 		var lowerIndex = 0;
204 		var upperIndex = sumRatings.length - 1;
205 		var middleIndex = 0;
206 		
207 		while (upperIndex > lowerIndex + 1) {
208 			middleIndex = Math.floor((lowerIndex + upperIndex) / 2);
209 			//alert("middleIndex = " + middleIndex);
210 			if (strictlyChronologicallyOrdered(sumRatings[middleIndex].getDate(), when)) {
211 				lowerIndex = middleIndex;
212 			}
213 			else{
214 				upperIndex = middleIndex;
215 			}
216 		}
217 		
218 		var previousRating = sumRatings[lowerIndex];
219 		var nextRating = sumRatings[upperIndex];
220 		
221 		if (strictlyEarlier) {
222 			return lowerIndex;
223 		}
224 		else {
225 			if(strictlyChronologicallyOrdered(when, nextRating.getDate())) {
226 				return lowerIndex;
227 			}
228 			else{
229 				return upperIndex;
230 			}
231 		}
232 	}

239 	}
