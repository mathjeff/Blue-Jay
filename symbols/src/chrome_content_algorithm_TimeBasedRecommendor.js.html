164     // reads one file and put its data into the TimeBasedRecommendor
165     // The data is expected to be in XML, and there is no error-checking
166     // Essentially this is an XML parser that creates Candidate, Rating, or Participation objects and adds them to the TimeBasedRecommendor
167     function readFile(fileName) {
168         //alert("recommendor reading file");
169         
170         
171         message("opening file " + fileName + "\r\n");
172 
173         // display a message
174         var fileContents = FileIO.readFile(fileName);
175         
176         message("initializing temporary variables");
177 
178 	    var currentChar;
179 	    var startTag = new Name();
180 	    var endTag = new Name();
181 	    var stackCount = 0;
182 	    var readingStartTag = false;
183 	    var readingValue = false;
184 	    var readingEndTag = false;
185         message("initializing local candidate");
186 	    var candidate = new Candidate();
187         message("initializing local name");
188 	    var value = new Name();
189 	    var objectName = new Name();
190         message("initializing local rating");
191 	    var rating = new Rating();
192 	    var activityType = new Name();
193         message("initializing local participation");
194 	    var participation = new Participation();
195     	
196 	    // setup some strings to search for in the file
197 	    var candidateIndicator = new Name("Candidate");
198 	    var nameIndicator = new Name("Name");
199 	    var parentIndicator = new Name("Parent");
200 	    var discoveryDateIndicator = new Name("DiscoveryDate");
201 
202         message("done initializing some temporary variables");
203         
204 	    var ratingIndicator = new Name("Rating");
205 	    var ratingActivityIndicator = new Name("Activity");
206 	    var ratingDateIndicator = new Name("Date");
207 	    var ratingValueIndicator = new Name("Score");
208 
209 	    var participationIndicator = new Name("Participation");
210 	    var participationActivityIndicator = new Name("Activity");
211 	    var participationStartDateIndicator = new Name("StartDate");
212 	    var participationEndDateIndicator = new Name("EndDate");
213 
214 	    var currentDate = new DateTime();
215 	    var characterIndex;
216 	    //alert("starting to parse file text");
217 	    // read until the file is finished
218 	    for (characterIndex = -1; characterIndex < fileContents.length; ) {
219 		    // Check if this is a new starttag or endtag
220 		    characterIndex++;
221 		    currentChar = fileContents[characterIndex];
222 		    
223 		    //message(currentChar);
224 		    // Check if this is the end of a tag
225 		    if (currentChar == '>') {
226 		        //message(value.getName());
227 			    //message(">\r\n");
228 			    characterIndex++;
229 			    currentChar = fileContents[characterIndex];
230 			    if (readingStartTag) {
231 				    if (stackCount == 1) {
232 					    // If we get here, then we just read the type of the object that is about to follow
233 					    objectName = startTag.makeCopy();
234 				    }
235 				    //message("start tag = ");
236 				    //message(startTag.getName() + ">");
237 				    value.clear();
238 				    readingValue = true;
239 			    }
240 			    if (readingEndTag) {
241 				    //message("end tag = ");
242 				    //message(endTag.getName() + ">\r\n");
243 				    if (stackCount == 2) {
244 					    // If we get here, then we just read an attribute of the object
245 
246 					    // If any of these trigger, then we just read an attribute of a Candidate (which is a song, artist, genre or whatever)
247 					    if (endTag.equalTo(nameIndicator))
248 						    candidate.setName(value);
249 					    if (endTag.equalTo(parentIndicator))
250 						    candidate.addParentName(value.makeCopy());
251 					    //if (endTag.equalTo(discoveryDateIndicator))
252 						//    candidate.setDiscoveryDate(new DateTime(value.getName()));
253 
254 
255 					    // If any of these trigger, then we just read an attribute of a rating
256 					    // Tags associated with ratings
257 					    if (endTag.equalTo(ratingActivityIndicator))
258 						    rating.setActivityName(value);
259 					    if (endTag.equalTo(ratingDateIndicator)) {
260 						    // keep track of the latest date ever encountered
261 						    //message("assigning date to rating\r\n");
262 						    currentDate = new DateTime(value.getName());
263 						    if (strictlyChronologicallyOrdered(latestDate, currentDate))
264 							    latestDate = currentDate;
265 						    rating.setDate(currentDate);
266 						    //message("done assigning date to rating\r\n");
267 					    }
268 					
269 					    if (endTag.equalTo(ratingValueIndicator)) {
270 						    rating.setScore(parseFloat(value.getName()));
271 					    }
272 					 
273 
274 					    // If any of these trigger, then we just read an attribute of a participation (an instance of listening)
275 					    if (endTag.equalTo(participationStartDateIndicator)) {
276 						    // keep track of the latest date ever encountered
277 						    currentDate = new DateTime(value.getName());
278 						    if (strictlyChronologicallyOrdered(latestDate, currentDate))
279 							    latestDate = currentDate;
280 						    participation.setStartTime(currentDate);
281 					    }
282 					    if (endTag.equalTo(participationEndDateIndicator)) {
283 						    // keep track of the latest date ever encountered
284 						    currentDate = new DateTime(value.getName());
285 						    if (strictlyChronologicallyOrdered(latestDate, currentDate))
286 							    latestDate = currentDate;
287 						    participation.setEndTime(currentDate);
288 					    }
289 					    if (endTag.equalTo(participationActivityIndicator))
290 						    participation.setActivityName(value.makeCopy());
291 				    }
292 				    if (stackCount == 1) {
293                         //alert("probably read a candidate");
294                         //message("object name = " + objectName.getName());
295 					    // If we get here then we just finished reading an object
296 					    if (objectName.equalTo(candidateIndicator)) {
297 						    // If we get here then we just finished reading a Candidate (which is a song, artist, genre or whatever)
298 						    // add the candidate to the inheritance hierarchy
299 						    //alert("adding candidate");
300 						    this.addCandidate(candidate);
301 						    candidate = new Candidate();
302 					    }
303 					    if (objectName.equalTo(ratingIndicator)) {
304 						    // If we get here then we just finished reading a rating
305 						    // add the rating to the rating set
306 						    this.putRatingInMemory(rating);
307 						
308 						    rating = new Rating();
309 					    }
310 					    if (objectName.equalTo(participationIndicator)) {
311 						    // If we get here then we just finished reading a rating
312 						    this.putParticipationInMemory(participation);
313 						    participation = new Participation();
314 					    }
315 				    }
316 				    stackCount--;
317 			    }
318 			    readingStartTag = false;
319 			    readingEndTag = false;
320 		    }
321 		    if (currentChar == '<') {
322 			    readingValue = false;
323 			    characterIndex++;
324 			    currentChar = fileContents[characterIndex];
325 			    if (currentChar != '/') {
326 			        //message("<");
327 				    // If we get here, then it's a start tag
328 				    startTag.clear();
329 				    stackCount++;
330 				    //message("stackCount = " + stackCount);
331 				    readingStartTag = true;
332 			    } else {
333 			        //message(value.getName() + "</");
334 				    // If we get here, then it's an end tag
335 				    endTag.clear();
336 				    readingEndTag = true;
337 			        characterIndex++;
338 			        currentChar = fileContents[characterIndex];
339 			    }
340 		    }
341 		    //message("start tag = ");
342 		    //message(startTag.getName());
343 		    // update names accordingly
344 		    if (readingStartTag) {
345 			    startTag.appendChar(currentChar);
346 		    }
347 		    if (readingValue) {
348 			    value.appendChar(currentChar);
349 		    }
350 		    if (readingEndTag) {
351 			    endTag.appendChar(currentChar);
352 		    }
353 	    }
354 	    //alert("done reading file " + fileName + "\r\n");
355     }

420     // returns the earliest date at which something happened. This is used as an estimate for the date that Bluejay first started running
421     function getEarliestInteractionDate() {
422         //alert("getting earliest interaction date");
423         if (earliestInteractionDate) {
424             // if we have a date at which something happened, return it
425             return earliestInteractionDate;
426         } else {
427             // If we don't have a date at which something happened, this is probably the first run
428             // The latest date is probably now, the startup date
429             var now = new DateTime();
430             now.setNow();
431             return now;
432         }

511     // inform each candidate of the categories it directly inherits from

513     function updateChildPointers() {
514         message("updating child pointers");
515 	    // iterate over each candidate
516 	    var candidateIterator = Iterator(candidates, true);
517 	    var currentCandidate;
518 	    var parent;
519 	    var i;
520 	    for (candidateName in candidateIterator) {
521 	        message("candidateName = " + candidateName);
522 		    currentCandidate = candidates[candidateName];
523 		    message("current candidate = ");
524 		    //message(currentCandidate);
525 		    message("current candidate is named" + currentCandidate.getName().getName());
526 		    //message("checking if update is needed\r\n");
527 		    if (currentCandidate.needToUpdateParentPointers()) {
528     		    message("update is required\r\n");
529 			    // if we get here then this candidate was added recently and its parents need their child pointers updated
530 			    var parentNames = currentCandidate.getParentNames();
531 			    for (i = 0; parentNames[i]; i++) {
532 		            message("parent name = " + parentNames[i].getName() + "\r\n");
533 				    parent = getCandidateWithName(parentNames[i]);
534 				    // don't add itself as a parent
535 				    if (currentCandidate != parent) {
536 				        message("parent is named " + parent.getName().getName() + "\r\n");
537 				        message("adding parent\r\n");
538 				        currentCandidate.addParent(parent);
539 				        message("adding child\r\n");
540 				        parent.addChild(currentCandidate);
541 				        message("done adding child\r\n");
542                     }
543 			    }
544 		    } else {
545 		        message("update was not required\r\n");
546 		    }
547 	    }
548 	    // save in "leafVector" all candidates that have no children
549 	    leafVector.length = 0;
550 	    candidateIterator = Iterator(candidates);
551 	    for ([candidateName, currentCandidate] in candidateIterator) {
552 	        if (currentCandidate.isPlayable()) {
553 	            leafVector.push(currentCandidate);
554 	        }
555 	    }
556     }

691     function estimateDiscoveryDate(candidate) {
692         // We assume here that the parents of a Candidate must have existed whenever the Candidate existed
693         //      (If we wanted to, we could simply record whenever anything was discovered
694         //      That would take more disk space and it would slow down the initial run when we have to record it for each song)
695         // We can simply record date that Bluejay first started up as the discovery date for the Candidate that contains all songs
696         // Our assumption tells us that the Candidate cannot be any older than the newest of its parents
697         var parents = candidate.getParents();
698         var i;
699         var tempDate;
700         // get a dateTime equal to now
701         var latestDate = new DateTime();
702         tempDate.setNow();
703         var currentParent;
704         // find the most-recently-discovered parent
705         for (i = 0; i < parents.length; i++) {
706             currentParent = parents[i];
707             // make sure that the parent's discovery date is correct
708             estimateDiscoveryDate(currentParent);
709             // get the parent's discovery date and compare
710             tempDate = currentParent.getDiscoveryDate();
711             if (strictlyChronologicallyOrdered(latestDate, tempDate)) {
712                 latestDate = tempDate;
713             }
714         }
715         // if the candidate thinks it was discovered longer ago than its most recent parent...
716         if (strictlyChronologicallyOrdered(latestDate, candidate.getDiscoveryDate())) {
717             // ...then tell the Candidate that it is probably wrong
718             candidate.suspectDiscoveryDate(latestDate);
719         }

794                 // check if they are in the correct order
795                 if (candidatesByScore[i].getCurrentRating().getMean() < candidatesByScore[i + deltaIndex].getCurrentRating().getMean()) {

798                     candidatesByScore[i] = candidatesByScore[i + deltaIndex];
799                     candidatesByScore[i + deltaIndex] = tempCandidate;
800                 }
801             }
802             // choose a smaller jump size for next time and continue
803             deltaIndex = Math.floor(deltaIndex / 2);
804         }
805         message("Scores after sorting:\r\n");
806         for (i = 0; i < candidatesByScore.length; i++) {

1127     			        outputWeight = count = sumWeight = 0.0;
1128     			    }
1129     			}
1130     			// Figure out whether we care about this distribution or not
1131 			    if ((stdDev == 0) || (!stdDevIsZero)) {
1132 				    // get the values from the distribution
1133 				    y = currentDistribution.getMean();
1134 				    if (stdDev == 0.0) {
1135 					    // If stddev is zero, then just use the given weight
1136 					    weight = currentDistribution.getWeight();
1137 				    } else {
1138 					    // If stddev is nonzero then weight based on both the stddev and the given weight
1139 					    weight = currentDistribution.getWeight() / stdDev;
1140 				    }
1141 				    // add to the running totals
1142 				    sumY += y * weight;
1143 				    sumY2 += y * y * weight;
1144 				    sumWeight += weight;
1145 				    sumVariance += stdDev * stdDev * weight;
1146 				    outputWeight += currentDistribution.getWeight();
1147 				    count += 1.0;
1148 			    }
1149 	        }
1150 	    }
1151 	    var result;
1152 	    if (sumWeight == 0) {
1153 	        result = new Distribution(.5, .5, 0);
1154 	    }
1155 	    else{
1156 		    // If we did have a distribution to predict from then we can calculate the average and standard deviations
1157 		    var newAverage = sumY / sumWeight;
1158 		    var variance1 = (sumY2 - sumY * sumY / sumWeight) / sumWeight;
1159 		    message("variance1 = ");
1160 		    message(variance1);
1161 		    message("\r\n");
1162 		    var variance2 = sumVariance / sumWeight;
1163 		    message("variance2 = ");
1164 		    message(variance2);
1165 		    message("\r\n");
1166 		    //stdDev = Math.sqrt(variance1 + variance2);
1167 		    stdDev = Math.sqrt(variance2);
1168 		    result = new Distribution(newAverage, stdDev, outputWeight);
1169 	    }
1170 	    
1171 	    message("resultant distribution = ");
1172 	    printDistribution(result);
1173 	    message("average of all distributions:" + (sumY / sumWeight) + "\r\n");
1174     	return result;
1175     }
