 20 messageToWrite = "";
 21 /** @class This is a class that handles reading and writing a file */
 22 FileIO = {
 23     
 24     /** Public Methods */
 25 
 26     /** read the file and return a string containing the file contents */
 27     readFile : function(fileName) {
 28         var homeDirFile = FileIO.getHomeDirectory();
 29         var file = homeDirFile;
 30         file.append(fileName);
 31         alert("Bluejay reading file: " + file.path);
 32 		
 33         /** open an input stream from the file */
 34         var istream = Components.classes["@mozilla.org/network/file-input-stream;1"].
 35                       createInstance(Components.interfaces.nsIFileInputStream);
 36         istream.init(file, 0x01, 0444, 0);
 37         
 38         // setup a converter to fill in a default character for anything we can't read 
 39         var converter = Components.classes["@mozilla.org/intl/converter-input-stream;1"].
 40                 createInstance(Components.interfaces.nsIConverterInputStream);
 41         var replacementCharacter = Components.interfaces.nsIConverterInputStream.DEFAULT_REPLACEMENT_CHARACTER
 42         converter.init(istream, "UTF-8", 0, replacementCharacter);
 43         var stringContents = "";
 44         var tempString = {};
 45         
 46         // read the file now
 47         while (converter.readString(4096, tempString) != 0) {
 48             stringContents += tempString.value;
 49         }
 50 
 51         converter.close(); /** this closes istream */
 52 
 53         return stringContents;
 54     },
 55     
 56     /**
 57      * Writes the given string to the file. "append" tells
 58      * whether to append the data or clear the file first
 59      */
 60     writeFile : function(fileName, stringData, append) {
 61         homeDirFile = FileIO.getHomeDirectory();
 62 
 63         var file = homeDirFile;
 64         file.append(fileName);
 65         /** alert("writing file " + file.path); */
 66         
 67         var data = stringData
 68         /** file is nsIFile, data is a string */
 69         var fileStream = Components.classes["@mozilla.org/network/file-output-stream;1"].
 70                        createInstance(Components.interfaces.nsIFileOutputStream);
 71         /** alert("writing files part 1"); */
 72          
 73         /**
 74          * use 0x02 | 0x10 to open file for appending.
 75          * File_io_flags 	
 76          *     0�01 	Read only
 77          *     0�02 	Write only
 78          *     0�04    Read and Write
 79          *     0�08 	Create File
 80          *     0�10 	Append
 81          *     0�20 	Truncate
 82          *     0�40 	Sync
 83          *     0�80 	Exclude
 84          */
 85 	    if (append)
 86             fileStream.init(file, 0x02 | 0x08 | 0x10, 0666, 0);
 87         else
 88             fileStream.init(file, 0x02 | 0x08 | 0x20, 0666, 0);
 89         
 90         /**
 91          * write, create, truncate.  In a c file operation, we have no
 92          * need to set file mode with or operation, directly using "r"
 93          * or "w" usually.  if you are sure there will never ever be any
 94          * non-ascii text in data you can also call fileStream.writeData
 95          * directly
 96          */
 97         var converter = Components.classes["@mozilla.org/intl/converter-output-stream;1"].
 98                         createInstance(Components.interfaces.nsIConverterOutputStream);
 99         converter.init(fileStream, "UTF-8", 0, 0);
100         converter.writeString(data);
101         converter.close(); /** this closes fileStream */
102 
103         /** alert("done writing file " + fileName); */
104     },
105 
106 	/** Private Methods */
107     getHomeDirectory : function() {
108         /** This will choose the user's home directory */
109         var dirService = Components.classes["@mozilla.org/file/directory_service;1"].
110                          getService(Components.interfaces.nsIProperties);
111         var homeDirFile = dirService.get("Home", Components.interfaces.nsIFile); /** returns an nsIFile object */
112         return homeDirFile;
113     }
114 };

116 // tells whether a message with the given priority should be displayed
117 function shouldLogMessagePriority(priority) {
118     if (priority > 0) {
119         return true;
120     } else {
121         return false;
122     }
123 }
124 /**
125  * the text is the message to send.  The priority determines
126  * whether it's a debug message or an important one
127  */
128 function message(text, priority) {
129     /** supply a default priority of 0 if none is provided */
130     if (!priority) {
131         priority = 0;
132     }
133     /**
134      * only save messages that we care about
135      * If we're not debugging, then don't include debug messages
136      */
137     if (shouldLogMessagePriority(priority)) {
138         // append the text to the end of the output file
139         //FileIO.writeFile("bluejay_output.txt", text, 1);
140         messageToWrite += text;
141     }
142 };
143 
144 /** append the text to the end of the output file */
145 function flushMessage() {
146     if (messageToWrite.length > 0) {
147         FileIO.writeFile("bluejay_output.txt", messageToWrite, 1);
148         messageToWrite = "";
149     }
150 };
