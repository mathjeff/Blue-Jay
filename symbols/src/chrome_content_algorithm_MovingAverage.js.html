 20 /**
 21  * @class the MovingAverage class represents the progression of 
 22  * values of some attribute of a Candidate over time
 23  */
 24 function MovingAverage() {
 25 	/** Private Member Variables */
 26 	var name;
 27 	var ownerName;
 28 
 29 	/** Public Methods */
 30       /**
 31        * returns a distribution of the expected values at this time, and
 32        * an integer identifying how many data points came before it
 33        */
 34 	this.getValueAt = getValueAt;
 35 	this.getCurrentValue = getCurrentValue;
 36 	
 37 	/** the vector returned is the relevant datapoints and the double
 38        * is the additional weight contributed by these datapoints
 39        */
 40 	this.getCorrelationsFor = getCorrelationsFor;
 41 	this.setName = setName;
 42 	this.getName = getName;
 43 	
 44 	/** the name of the Candidate that this MovingAverage describes */
 45 	this.setOwnerName = setOwnerName;
 46 	this.getOwnerName = getOwnerName;
 47 	
 48 	/**
 49        * for determining if its type is ParticipationMovingAverage or not
 50        */
 51 	this.isAParticipationMovingAverage = isAParticipationMovingAverage;		
 52 	this.getLatestDate = getLatestDate;
 53 	this.stringVersion = stringVersion;
 54     /** function definitions */
 55     
 56     /**
 57      * a MovingAverage doesn't know how to return its value. This
 58      * function must be overridden
 59      */
 60     function getValueAt(when, strictlyEarlier) {
 61         alert("MovingAverage::getValueAt() was called. This is an error.");
 62 		
 63         /** setup an invalid distribution */
 64 	    var distribution = new Distribution(0, 0, 0);
 65 		
 66 	  /**
 67          * return an array with the invalid
 68          * distribution and invalid index
 69          */
 70 	    return [distribution, -1];
 71     }
 72 	
 73     /** gets the current value of whatever variable we're tracking */
 74     function getCurrentValue(when, strictlyEarlier) {
 75         var resultArray = this.getValueAt(when, strictlyEarlier);
 76         var resultantDistribution = resultArray[0];
 77         return resultantDistribution;
 78     }
 79 	
 80     /**
 81      * makes a bunch of datapoints that describe how the value of the
 82      * 'other' distribution changes with this one
 83      */
 84     function getCorrelationsFor(other, previousNumRatings) {
 85         /** message("MovingAverage::getCorrelationsFor pt0"); */
 86         var i;
 87         var otherRatings = other.getRatings();
 88 
 89         var results = [];
 90         /**
 91          * count how many individual x-values will
 92          * be used to create the prediction
 93          */
 94         var numChanges = 0; 
 95     
 96         /** var startingIndex = other.getIndexForDate(startTime, false) + 1; */
 97         /** message("MovingAverage::getCorrelationsFor pt1"); */
 98         var startingIndex = previousNumRatings;
 99         if ((otherRatings.length > 0) && (previousNumRatings < otherRatings.length)) {
100             var startTime = otherRatings[previousNumRatings].getDate();
101             /** find the starting index. This can
102             * be optimized with a binary search! */
103             for (i = otherRatings.length - 1; i >= 0; i--) {
104                 if (strictlyChronologicallyOrdered(otherRatings[i].getDate(), startTime))
105                     break;
106             }
107             message("MovingAverage::getCorrelationsFor pt2");
108             var startingIndex = i + 1;
109             
110             var x, y, weight;
111             weight = 1;
112             var previousIndex = this.getValueAt(startTime, true)[1];
113             
114     		
115 		    
116             /**
117              * don't count the transition from "undefined"
118              * to "defined" as a change
119              */
120             if ((startingIndex == 0) || (previousIndex < 0))
121                 numChanges = -1;
122     		
123 	        /**
124                * This should be improved eventually.  We should give the
125                * deviation of each point to the scatterplot in some
126                * meaningful way
127                */
128               /** message("MovingAverage::getCorrelationsFor pt3"); */
129             for (i = startingIndex; i < otherRatings.length; i++) {
130                 var value = this.getValueAt(otherRatings[i].getDate(), true);
131                 x = value[0].getMean();
132                 y = otherRatings[i].getScore();
133                 var ourIndex = value[1];
134                 if (ourIndex >= 0) {
135                     /**message("MovingAverage::getCorrelationsFor pt4");*/
136                     if (ourIndex != previousIndex) {
137                         previousIndex = value[1];
138                         numChanges++;
139                     }
140                     results.push(new Datapoint(x, y, weight));
141                 }
142             }
143         }
144         /**
145          * if we had to skip a change but there were
146          * none to skip, then make it zero
147          */
148         if (numChanges < 0)
149             numChanges = 0;
150         return [results, numChanges];
151     }
152     function setName(newName) {
153         name = newName;
154     }
155     function getName() {
156         return name;
157     }
158     function setOwnerName(newName) {
159         this.ownerName = newName;
160     }
161     function getOwnerName() {
162         return this.ownerName;
163     }
164     function isAParticipationMovingAverage() {
165         alert("MovingAverage::isAParticipationMovingAverage() was called. This is an error.");
166         return false;
167     }
168     function getLatestDate() {
169         alert("MovingAverage::getLatestDate() was called. This is an error.");
170         return new DateTime();
171     }
172     function stringVersion() {
173         return "I am a MovingAverage";
174     }
175 };
